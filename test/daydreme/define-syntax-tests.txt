(
(andSyntax
    (begin
        (define-syntax and
            (syntax-rules ()
                ((_) #t)
                ((_ e) e)
                ((_ e1 e2 e3 ...)
                    (if e1 (and e2 e3 ...) #f))))
        (and #t #t #t))
    #t)
(orSyntax
    (begin
        (define-syntax or
            (syntax-rules ()
                ((_) #f)
                ((_ e) e)
                ((_ e1 e2 e3 ...)
                 (let ((t e1))
                    (if t t (or e2 e3 ...))))))
        (or #f #f (or #f (or) #t)))
    #t)
)
